Ticket Table creation schema:

-- Enable pgvector
CREATE EXTENSION IF NOT EXISTS vector;

-- Create tickets table
CREATE TABLE IF NOT EXISTS tickets (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  issue_key TEXT UNIQUE NOT NULL,

  name TEXT NOT NULL,
  email TEXT NOT NULL,

  summary TEXT NOT NULL,
  description TEXT,

  status TEXT NOT NULL DEFAULT 'Open'
    CHECK (status IN ('Open', 'Completed')),

  is_duplicate BOOLEAN DEFAULT FALSE,
  parent_ticket_key TEXT,

  -- sentence-transformers embedding
  embedding VECTOR(384),

  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Add all extra columns in one block
ALTER TABLE tickets
ADD COLUMN IF NOT EXISTS checklist_steps JSONB,
ADD COLUMN IF NOT EXISTS commands JSONB,
ADD COLUMN IF NOT EXISTS priority TEXT DEFAULT 'P5',
ADD COLUMN IF NOT EXISTS priority_label TEXT DEFAULT 'Planning',
ADD COLUMN IF NOT EXISTS sla_response_time TEXT,
ADD COLUMN IF NOT EXISTS sla_resolution_time TEXT,
ADD COLUMN IF NOT EXISTS runbook_title TEXT,
ADD COLUMN IF NOT EXISTS runbook_category TEXT,
ADD COLUMN IF NOT EXISTS match_type TEXT;



vector_index schema:

create index tickets_embedding_idx
on tickets
using ivfflat (embedding vector_cosine_ops)
with (lists = 100);



ticket_simlarity_srearch schema:

create or replace function match_tickets(
  query_embedding vector(384),
  match_count int
)
returns table (
  id uuid,
  issue_key text,
  summary text,
  similarity float
)
language sql
stable
as $$
  select
    id,
    issue_key,
    summary,
    1 - (embedding <=> query_embedding) as similarity
  from tickets
  where embedding is not null
  order by embedding <=> query_embedding
  limit match_count;
$$;
